\newcommand\rectanglepath[5][]{%
\scope[shift={(#2)},#1]%
\scope[overlay]
\node[font={\CJKfamily{zifont}\color{#4}}] at (0,0) {#5\rlap{\phantom{字}}};
\endscope
\draw[myred, line width=1pt] (-0.4,-0.4) rectangle ++(0.8,0.8);
\draw[myred, dashed] (-0.4,0.4) coordinate (#3-tl)
-- (0.4,-0.4) coordinate (#3-br);
\draw[red, dashed] (0.4,0.4) coordinate (#3-tr)
-- (-0.4,-0.4) coordinate (#3-bl);
\draw[myred, dashed] (0,0.4) coordinate (#3-t)
-- (0,-0.4) coordinate (#3-b);
\draw[myred, dashed] (-0.4,0) coordinate (#3-l)
-- (0.4,0) coordinate (#3-r);
\endscope}

    \endscope
}
